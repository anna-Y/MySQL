表字段     id name age hobby score comment

CREATE TABLE student(
id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(32) NOT NULL,
age INT NOT NULL,hobby SET('football','computer','running','basketball'),
score FLOAT DEFAULT 0.0,
comment TEXT);

查找
    所有年龄不到10岁或者大于14岁的同学
    SELECT * FROM student WHERE age<10 OR age>14;
    #课堂写法
    select * from student where age not between 10 and 14;
    兴趣爱好中包含computer的同学
    SELECT * FROM student WHERE hobby LIKE'%computer%';
    #课堂写法
    select * from student where find_in_set('computer',hobby);
    年龄大于等于15，又喜欢足球的同学
    SELECT * FROM student WHERE age>=15 AND hobby LIKE'%football%';
    #课堂写法
    select * from student where age>=15 AND find_in_set('football',hobby);
    不及格兴趣爱好又不为空的同学
    SELECT * FROM student WHERE score<60 AND hobby IS NOT NULL;
    #课堂写法
    select * from student where score<60 and hobby is not null;
    成绩大于90分的所有同学，只看姓名和成绩
    SELECT name,score FROM student WHERE score>90;
    #课堂写法
    select name,score from student where score>90;

作业： 1. 熟练MySQL基本语句：
                            创建数据库   CREATE DATABASE 库名;
                            创建数据表   CREATE TABLE 表名;
        插入，     insert into 表名 values();
        查询，     select *全部字段名/(查询字段名) from 表名 where 查询条件;（查询指定部分）
        修改，     update 表名 set 修改字段 where 修改条件;（修改指定部分）
        删除，     delete from 表名 where 删除条件;（删除指定部分）
        表结构修改   alter table 表名 修改动作;(add/drop/modify/change/rename)
